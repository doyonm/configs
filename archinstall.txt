Install Archlinux with archinstall script using minimal setup

Additional packages

xf86-video-fbdev xorg xorg-xinit base-devel neofetch vim git picom feh alacritty rofi rust thunar mc avahi nss-mdns openssh unzip wget thunar-archive-plugin file-roller gtk3

reboot and login to sudoer account

mkdir git && cd git

# Leftwm Theme
git clone https://github.com/doyonm/TYSLWMTheme.git
# Config files
git clone https://github.com/doyonm/configs.git

# Install AUR packages (makepkg -si)
git clone https://aur.archlinux.org/leftwm.git
git clone https://aur.archlinux.org/polybar.git
git clone https://aur.archlinux.org/brave-bin.git

cp /etc/X11/xinit/xinitrc .xinitrc
vim .xinitrc
# comment last 5 lines (xwm, xclock xterm)
picom &
exec leftwm

# Set Locale
vim /etc/locale.gen
locale-gen

vim /etc/locale.conf
LANG=fr_CA.UTF-8

# Install Fonts
sudo pacm -S noto-fonts noto-fonts-extra adobe-source-code-pro-fonts cantarell-fonts gnu-free-fonts gsfonts ttf-dejavu
fc-cache

# Hack fonts
wget https://github.com/source-foundry/Hack/releases/download/v3.003/Hack-v3.003-ttf.zip
unzip Hack-v3.003-ttf.zip

# leftwm config
mkdir .config/leftwm/themes

cd .config/leftwm/themes
ln -s /home/doyonm/git/TYSLWMTheme current

startx

# Leftwm additionals key binding
Shit+Super <Enter> (Open alacritty)
Shit+Super <Left> | <Right> Resize Master View
Super <b> (Open Brave)
Super <f> (Open thunar)
